\documentclass[12pt]{article}
\usepackage{cu-calendar}
\begin{document}

\makeatletter
\def\assertSame#1#2{%
    \edef\@aa{#1}%
    \edef\@bb{#2}%
    \ifx\@aa\@bb\else
        \errmessage{TEST FAILED: #1: "\@aa" expected "\@bb"}%
    \fi
}%

\assertSame{\cu@jdnFromJulianDate{2016}{3}{30}}{2457491}
\assertSame{\cu@jdnFromGregorianDate{2016}{3}{30}}{2457478}

\assertSame{\cu@calendarJulian{2457491}}{2016-3-30}
\assertSame{\cu@calendarGregorian{2457478}}{2016-3-30}

\assertSame{\cu@julianFromGregorian{2016}{3}{30}}{2016-3-17}
\assertSame{\cu@gregorianFromJulian{2016}{3}{17}}{2016-3-30}

\assertSame{\cuJulianDate{2016}{3}{30}}{17 ма́рта л. 2016}
\assertSame{\cuGregorianDate{2016}{3}{30}}{30 ма́рта л. 2016}

\makeatother

\end{document}
